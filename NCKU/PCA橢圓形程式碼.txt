library(factoextra)
a <- data.matrix(iris[-5])

# center: whether the variables should be shifted to be zero centered.
# scale.: whether the variables should be scaled to have unit variance, in general scaling is advisable.
b <- prcomp(a, scale. = TRUE, center = TRUE) 

# axes: pricipal component to be plotted.
# col.ind: color for individuals and variables
fviz_pca_ind(b, axes = c(1,3), geom = c("point"), ellipse.level = 0.95,
             col.ind = iris$Species,
             addEllipses = TRUE)
